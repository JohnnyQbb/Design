    <div class="BH-frame-autolayout-align_center BH-select">
        <select>
            <option value="0">default</option>
            <option value="1">A</option>
            <option value="2">B</option>
            <option value="3">C</option>
        </select>
    </div>
    <script>
        var x, i, j, l, ll, selElmnt, a, b, c;
        x = document.getElementsByClassName("BH-select");
        l = x.length;
        for (i = 0; i < l; i++) {
            selElmnt = x[i].getElementsByTagName("select")[0];
            ll = selElmnt.length;
            a = document.createElement("DIV");
            a.setAttribute("class", "select-selected fa");
            a.innerHTML = selElmnt.options[selElmnt.selectedIndex].innerHTML;
            x[i].appendChild(a);
            b = document.createElement("DIV");
            b.setAttribute("class", "select-items select-hide");
            for (j = 1; j < ll; j++) {
                c = document.createElement("DIV");
                c.innerHTML = selElmnt.options[j].innerHTML;
                c.addEventListener("click", function (e) {
                    var y, i, k, s, h, sl, yl;
                    s = this.parentNode.parentNode.getElementsByTagName("select")[0];
                    sl = s.length;
                    h = this.parentNode.previousSibling;
                    for (i = 0; i < sl; i++) {
                        if (s.options[i].innerHTML == this.innerHTML) {
                            s.selectedIndex = i;
                            h.innerHTML = this.innerHTML;
                            y = this.parentNode.getElementsByClassName("same-as-selected");
                            yl = y.length;
                            for (k = 0; k < yl; k++) {
                                y[k].removeAttribute("class");
                            }
                            this.setAttribute("class", "same-as-selected");
                            break;
                        }
                    }
                    h.click();
                });
                b.appendChild(c);
            }
            x[i].appendChild(b);
            a.addEventListener("click", function (e) {
                e.stopPropagation();
                closeAllSelect(this);
                this.nextSibling.classList.toggle("select-hide");
                this.classList.toggle("select-active");
            });
        }
    
        function closeAllSelect(elmnt) {
            var x, y, i, xl, yl, arrNo = [];
            x = document.getElementsByClassName("select-items");
            y = document.getElementsByClassName("select-selected");
            xl = x.length;
            yl = y.length;
            for (i = 0; i < yl; i++) {
                if (elmnt == y[i]) {
                    arrNo.push(i)
                } else {
                    y[i].classList.remove("select-active");
                }
            }
            for (i = 0; i < xl; i++) {
                if (arrNo.indexOf(i)) {
                    x[i].classList.add("select-hide");
                }
            }
        }
        document.addEventListener("click", closeAllSelect);
    </script>

        <div id='menuBar' class="bar"></div>
    </ul>
    <script>
        const menuTab = document.getElementById("menuTab");
        const menuBar = document.getElementById("menuBar");
    
        // Set the initial position and width of the bar
        const isActive = menuTab.querySelector('.is-active');
        menuBar.style.transform = `translateX(${isActive.offsetLeft}px)`;
        menuBar.style.width = `${isActive.offsetWidth}px`;
    
        // Added an event listener for the mouseover event on the ul element
        menuTab.addEventListener("mouseover", function (event) {
            // Check if the target element is an li element
            const target = event.target.closest("li");
            if (target) {
                // Set the position and width of the bar based on the target li element
                menuBar.style.transform = `translateX(${target.offsetLeft}px)`;
                menuBar.style.width = `${target.offsetWidth}px`;
            }
        });
    
        // Added an event listener for the mouseleave event on the ul element
        menuTab.addEventListener("mouseleave", function (event) {
            // Set the position and width of the bar back to the initial li element with class "is-active"
            menuBar.style.transform = `translateX(${isActive.offsetLeft}px)`;
            menuBar.style.width = `${isActive.offsetWidth}px`;
        });
    </script>
